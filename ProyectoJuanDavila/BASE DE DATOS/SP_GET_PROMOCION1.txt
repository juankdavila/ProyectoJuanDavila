-- FUNCTION: public.SP_GET_PROMOCION(integer)

-- DROP FUNCTION IF EXISTS public."SP_GET_PROMOCION"(integer);

CREATE OR REPLACE FUNCTION public."SP_GET_PROMOCION"(
	pi_id_promocion integer)
    RETURNS void
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
AS $BODY$
BEGIN
	
	SELECT * FROM "PROMOCIONES"
	WHERE ID_PROMOCION=PI_ID_PROMOCION;

END;
$BODY$;

ALTER FUNCTION public."SP_GET_PROMOCION"(integer)
    OWNER TO postgres;
