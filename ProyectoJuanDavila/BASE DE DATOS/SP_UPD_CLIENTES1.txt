-- FUNCTION: public.SP_UPD_CLIENTES(character varying, character varying)

-- DROP FUNCTION IF EXISTS public."SP_UPD_CLIENTES"(character varying, character varying);

CREATE OR REPLACE FUNCTION public."SP_UPD_CLIENTES"(
	pv_cedula character varying,
	pv_nombres character varying)
    RETURNS void
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
AS $BODY$
BEGIN
	
	UPDATE "CLIENTES"
	SET NOMBRE=PV_NOMBRES
	WHERE CEDULA=PV_CEDULA;

END;
$BODY$;

ALTER FUNCTION public."SP_UPD_CLIENTES"(character varying, character varying)
    OWNER TO postgres;
