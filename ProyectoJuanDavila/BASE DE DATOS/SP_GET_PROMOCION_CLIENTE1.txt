-- FUNCTION: public.SP_GET_PROMOCION_CLIENTE(character varying)

-- DROP FUNCTION IF EXISTS public."SP_GET_PROMOCION_CLIENTE"(character varying);

CREATE OR REPLACE FUNCTION public."SP_GET_PROMOCION_CLIENTE"(
	pv_cedula character varying)
    RETURNS void
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
AS $BODY$
BEGIN
	
	SELECT isnull(SUM(MONTO_DESCUENTO),0) MONTO_DESCUENTO
	FROM "PROMOCIONES_CLIENTES"
	WHERE CEDULA=PV_CEDULA;

END;
$BODY$;

ALTER FUNCTION public."SP_GET_PROMOCION_CLIENTE"(character varying)
    OWNER TO postgres;
