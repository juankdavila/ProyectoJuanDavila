-- FUNCTION: public.SP_GET_HISTORIAL_VISITA(character varying)

-- DROP FUNCTION IF EXISTS public."SP_GET_HISTORIAL_VISITA"(character varying);

CREATE OR REPLACE FUNCTION public."SP_GET_HISTORIAL_VISITA"(
	pv_cedula character varying)
    RETURNS void
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
AS $BODY$
BEGIN
	
	SELECT B.DESCRIPCION,C.NOMBRE,A.FECHA_VISITA
FROM "VISITA_TIENDA_CLIENTES" A
INNER JOIN TIENDAS B ON A.ID_TIENDA=B.ID_TIENDA
INNER JOIN CLIENTES C ON A.CEDULA=C.CEDULA
WHERE A.CEDULA=PV_CEDULA
ORDER BY FECHA_VISITA DESC;

END;
$BODY$;

ALTER FUNCTION public."SP_GET_HISTORIAL_VISITA"(character varying)
    OWNER TO postgres;
