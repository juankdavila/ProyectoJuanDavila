-- FUNCTION: public.SP_GET_USUARIO(integer)

-- DROP FUNCTION IF EXISTS public."SP_GET_USUARIO"(integer);

CREATE OR REPLACE FUNCTION public."SP_GET_USUARIO"(
	pi_id_usuario integer)
    RETURNS SETOF "USUARIOS" 
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
    ROWS 1000

AS $BODY$
BEGIN
    RETURN QUERY 
	SELECT * FROM "USUARIOS" WHERE ID_USUARIO = PI_ID_USUARIO;
END;
$BODY$;

ALTER FUNCTION public."SP_GET_USUARIO"(integer)
    OWNER TO postgres;
