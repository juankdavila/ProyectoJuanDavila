-- FUNCTION: public.SP_GET_CLIENTE(character varying)

-- DROP FUNCTION IF EXISTS public."SP_GET_CLIENTE"(character varying);

CREATE OR REPLACE FUNCTION public."SP_GET_CLIENTE"(
	pv_cedula character varying)
    RETURNS void
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
AS $BODY$
BEGIN
	
	SELECT * FROM "CLIENTES"
	WHERE CEDULA=PV_CEDULA;

END;
$BODY$;

ALTER FUNCTION public."SP_GET_CLIENTE"(character varying)
    OWNER TO postgres;
