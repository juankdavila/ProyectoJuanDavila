-- FUNCTION: public.SP_DEL_CLIENTES(character varying)

-- DROP FUNCTION IF EXISTS public."SP_DEL_CLIENTES"(character varying);

CREATE OR REPLACE FUNCTION public."SP_DEL_CLIENTES"(
	pv_cedula character varying)
    RETURNS void
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
AS $BODY$
BEGIN
	
	UPDATE "CLIENTES"
	SET estado = 'I'
	WHERE cedula= pv_cedula;  

END;
$BODY$;

ALTER FUNCTION public."SP_DEL_CLIENTES"(character varying)
    OWNER TO postgres;
