-- FUNCTION: public.SP_UPD_TIENDAS(integer, character varying, character varying)

-- DROP FUNCTION IF EXISTS public."SP_UPD_TIENDAS"(integer, character varying, character varying);

CREATE OR REPLACE FUNCTION public."SP_UPD_TIENDAS"(
	pv_id_tienda integer,
	pv_descripcion character varying,
	pv_estado character varying)
    RETURNS void
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
AS $BODY$
BEGIN
	
	UPDATE "TIENDAS"
	SET DESCRIPCION=PV_DESCRIPCION,
    ESTADO=PV_ESTADO
	WHERE ID_TIENDA=PV_ID_TIENDA;

END;
$BODY$;

ALTER FUNCTION public."SP_UPD_TIENDAS"(integer, character varying, character varying)
    OWNER TO postgres;
